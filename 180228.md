function findDulicates (arr) {
	if (arr.length){
		return [];
	}

	var result = [];
	var storage = {};

	for (var i = 0 ; i < arr.length; i++){
		if (!storage[arr[i]]){
			storage[arr[i]]=true;
		} else {
			result.push(arr[i]);	
		}
		
	}

	return result;
}
